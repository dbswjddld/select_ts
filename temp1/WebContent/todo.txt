create or replace PROCEDURE p_ts_list (t_list out SYS_REFCURSOR)
IS
BEGIN
  OPEN t_list FOR
    select a.tablespace_name, a.status, sum(b.bytes/1024/1024) as total, sum(c.bytes/1024/1024) as free
    from dba_tablespaces a, dba_data_files b, dba_free_space c
    where a.tablespace_name = b.tablespace_name
      and a.tablespace_name = c.tablespace_name
    group by a.tablespace_name, a.status;
END p_ts_list;
